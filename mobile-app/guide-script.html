uiJSAlert

uiJSAlert(options);

디자인 얼럿 함수
사용 예제는 [공통 > 얼럿 레이어 팝업] 화면 참고

options : object : 얼럿 옵션 오브젝트

options.title : string : 타이틀 텍스트 : default = ''

options.message : string : 메세지 텍스트 : default = ''

options.buttons : array : 버튼 배열 : default = [{}]

options.buttons[0].text : string : 버튼 텍스트 : default = '확인'

options.buttons[0].type : string : 버튼 스타일 (secondary|tertiary|quaternary) : default = ''

options.buttons[0].callback : function : 버튼을 클릭 했을때 실행 할 함수 : default = function (closeFn) { closeFn(); }

options.buttons[0].html : function : 버튼 html 만드는 함수
// options.buttons[0].html default

function (options, triggerClassName) {
  var html = '';
  var type = options.type.length ? 'ui-basic-button--' + options.type : '';

  html += <button type="button" class="ui-button ui-basic-button ui-basic-button--regular ' + type + ' ' + triggerClassName + '">'; // triggerClassName 값으로 해당 요소를 클릭 요소로 지정
  html += '<span class="ui-basic-button__text">' + options.text + '</span>';
  html += '</button>';

  return html;
}
